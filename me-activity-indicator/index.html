meActivityIndicator

Notificaciones tipo Dynamic Island para la web. Zero dependencias. Spring physics. Accesible.

v3.0 MIT Zero deps

Inicialización

const indicator = new meActivityIndicator({
  position: 'top-center',
  theme: 'dark',
  typeColors: {
    premium: '#8b5cf6',
    light: '#facc15'
  },
  icons: {
    premium: '<svg ...>...</svg>',
    heart: '<svg ...>...</svg>'
  }
});

Config Helpers

// Cambiar posición
indicator.setPosition('top-center');
// Opciones: 'top-left', 'top-center', 'top-right',
//           'center', 'bottom-left', 'bottom-center', 'bottom-right'

// Cambiar tema
indicator.setTheme('dark');
// Opciones: 'dark', 'light', 'system'

// Cambiar estilo de apilamiento
indicator.setStackStyle('3d');
// Opciones: '3d', 'fan', 'counter'

// Cambiar ancho de la isla
indicator.setIslandWidth('normal');
// Opciones: 'compact', 'normal', 'wide'
